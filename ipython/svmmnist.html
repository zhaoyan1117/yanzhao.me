Grids generator¶

In [8]:
def grids_generator(image, grid_size):
    row_size, col_size = image.shape
    half_grid_size = grid_size / 2

    row_start_index = 0
    row_end_index = row_start_index + grid_size - 1
    while row_end_index < row_size:
        col_start_index = 0
        col_end_index = col_start_index + grid_size - 1
        while col_end_index < col_size:
            yield image[row_start_index:row_end_index,
                        col_start_index:col_end_index]
            col_start_index += half_grid_size
            col_end_index += half_grid_size
        row_start_index += half_grid_size
        row_end_index += half_grid_size

Get hog function¶

In [13]:
def get_hog(image, *grid_sizes, **params):
    normalize = params.get('normalize', True)
    gaussian = params.get('gaussian', False)
    sigma = params.get('sigma', 1)
    
    tap_filter = np.array([[-1], [0], [1]])
    
    if gaussian:
        image = gaussian_filter(image, sigma)
    
    vr = convolve2d(image, tap_filter, mode='same')
    hr = convolve2d(image, np.transpose(tap_filter), mode='same')
    orientations = np.degrees(np.arctan2(hr, vr))

    feature = []
    for size in grid_sizes:
        orientations_grids = grids_generator(orientations, size)

        for grid in orientations_grids:
            hist, _ = np.histogram(grid, bins=9, range=(-180.0, 180.0))
            if normalize:
                hist = hist / float(hist.sum())
            feature.append(hist)
    return np.array(feature).flatten()

# http://www.datawrangling.com/python-montage-code-for-displaying-arrays/
import sys
import os
import time

from numpy import array,flipud,shape,zeros,rot90,ceil,floor,sqrt
from scipy import io,reshape,size
 
def montage(X):    
    m, n, count = shape(X)    
    mm = int(ceil(sqrt(count)))
    nn = mm
    M = zeros((mm * m, nn * n)) + 255

    image_id = 0
    for j in range(mm):
        for k in range(nn):
            if image_id >= count: 
                break
            sliceM, sliceN = j * m, k * n
            M[sliceN:sliceN + n, sliceM:sliceM + m] = X[:, :, image_id]
            image_id += 1

    fig = plt.figure(figsize=(4,4))
    plt.imshow(flipud(rot90(M)), cmap='gray')
    plt.axis('off')
    plt.show();
    if save_fig:
        fig.savefig('{0}/montage.png'.format(fig_directory), dpi=fig_dpi)
